line -> symbol = fexpr | symbol[symbol symbollist] = fexpr | nexpr
symbollist -> eps | , symbol symbollist
fexpr -> symbol | symbol[fexpr fexprlist] | Id[number] | C[number] |
         Cn[fexpr fexprlist] | Pr[fexpr, fexpr] | Mn[fexpr] | S
fexprlist -> epe | , fexpr fexprlist
nexpr -> fexpr(nexprlist) | natural
nexprlist -> eps | nexpr nexprlist2
nexprlist2 -> eps | , nexpr nexprlist2
natural -> number | [naturallist]
naturallist -> eps | natural naturallist2
naturallist2 -> eps | , natural naturallist2
number -> digit*
